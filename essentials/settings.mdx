Every Mintlify site needs a `docs.json` file at the project root. This page documents the available properties.

<Info>Previously named `mint.json`, the configuration file was renamed to `docs.json`. If you still have a `mint.json`, rename it to `docs.json`.</Info>

## Core properties

<ResponseField name="name" type="string" required>
  Display name for your documentation site. Used in the browser tab title and SEO.

  Example: `"Zajaly Developer Docs"`
</ResponseField>

<ResponseField name="theme" type="string">
  The visual theme for your site. Options: `"willow"`, `"mint"`, `"palm"`, `"fern"`, `"prism"`.

  Example: `"willow"`
</ResponseField>

<ResponseField name="description" type="string">
  A short description used for SEO `<meta>` tags.
</ResponseField>

<ResponseField name="logo" type="string | object">
  Path to a logo image, or an object with light/dark variants.
  <Expandable title="Logo object">
    <ResponseField name="light" type="string">
      Path to the logo used in light mode.
    </ResponseField>
    <ResponseField name="dark" type="string">
      Path to the logo used in dark mode.
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      URL that the logo links to when clicked.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Path to the favicon file. Supports `.svg`, `.png`, and `.ico`.
</ResponseField>

<ResponseField name="colors" type="object">
  Hex color codes for your site theme.
  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      Primary accent color used for links, buttons, and highlights.
    </ResponseField>
    <ResponseField name="light" type="string">
      Accent color used specifically in dark mode.
    </ResponseField>
    <ResponseField name="dark" type="string">
      Color used for important buttons and CTAs.
    </ResponseField>
    <ResponseField name="background" type="object">
      Custom background colors for light and dark modes.
      <Expandable title="Background">
        <ResponseField name="light" type="string">Hex color for light mode background.</ResponseField>
        <ResponseField name="dark" type="string">Hex color for dark mode background.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Navigation

<ResponseField name="navigation" type="object" required>
  Controls sidebar tabs, groups, and page ordering. See [Navigation](/essentials/navigation) for full details.
  <Expandable title="Navigation">
    <ResponseField name="tabs" type="Tab[]">
      Array of sidebar tabs. Each tab contains groups of pages.
      <Expandable title="Tab">
        <ResponseField name="tab" type="string">Display label for the tab.</ResponseField>
        <ResponseField name="icon" type="string">Font Awesome icon name.</ResponseField>
        <ResponseField name="groups" type="Group[]">Array of page groups within this tab.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="global" type="object">
      Global navigation elements visible across all tabs.
      <Expandable title="Global">
        <ResponseField name="anchors" type="Anchor[]">
          External links shown in the sidebar.
          <Expandable title="Anchor">
            <ResponseField name="anchor" type="string">Label for the link.</ResponseField>
            <ResponseField name="icon" type="string">Font Awesome icon.</ResponseField>
            <ResponseField name="href" type="string">URL the anchor links to.</ResponseField>
          </Expandable>
        </ResponseField>
        <ResponseField name="tabs" type="GlobalTab[]">
          Tabs visible on every page (e.g., Changelog).
          <Expandable title="GlobalTab">
            <ResponseField name="tab" type="string">Display label.</ResponseField>
            <ResponseField name="icon" type="string">Font Awesome icon.</ResponseField>
            <ResponseField name="href" type="string">Internal path.</ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Navbar

<ResponseField name="navbar" type="object">
  Top navigation bar configuration.
  <Expandable title="Navbar">
    <ResponseField name="links" type="Link[]">
      Array of links shown in the top bar.
      <Expandable title="Link">
        <ResponseField name="type" type="string">`"github"` for a GitHub link with star count, or omit for a standard link.</ResponseField>
        <ResponseField name="href" type="string">URL for the link.</ResponseField>
        <ResponseField name="label" type="string">Display text (for non-GitHub links).</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="primary" type="object">
      Primary call-to-action button in the navbar.
      <Expandable title="Primary CTA">
        <ResponseField name="type" type="string">`"button"` or `"link"`.</ResponseField>
        <ResponseField name="label" type="string">Button text.</ResponseField>
        <ResponseField name="href" type="string">URL the button links to.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Footer

<ResponseField name="footer" type="object">
  Footer configuration with social links and link columns.
  <Expandable title="Footer">
    <ResponseField name="socials" type="object">
      Key-value pairs of social platform and URL. Supported keys: `website`, `facebook`, `x`, `discord`, `slack`, `github`, `linkedin`, `instagram`, `hacker-news`.
    </ResponseField>
    <ResponseField name="links" type="LinkColumn[]">
      Array of link columns displayed in the footer.
      <Expandable title="LinkColumn">
        <ResponseField name="header" type="string">Column heading.</ResponseField>
        <ResponseField name="items" type="LinkItem[]">
          <Expandable title="LinkItem">
            <ResponseField name="label" type="string">Link text.</ResponseField>
            <ResponseField name="href" type="string">URL.</ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Appearance and fonts

<ResponseField name="appearance" type="object">
  <Expandable title="Appearance">
    <ResponseField name="default" type="string">
      Default color mode for new visitors. `"light"` or `"dark"`.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="fonts" type="object">
  Custom fonts for headings and body text. Loaded from Google Fonts.
  <Expandable title="Fonts">
    <ResponseField name="heading" type="object">
      <Expandable title="Heading font">
        <ResponseField name="family" type="string">Google Font family name.</ResponseField>
        <ResponseField name="weight" type="number">Font weight (e.g., 400, 700).</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="body" type="object">
      <Expandable title="Body font">
        <ResponseField name="family" type="string">Google Font family name.</ResponseField>
        <ResponseField name="weight" type="number">Font weight.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## API configuration

<ResponseField name="openapi" type="string | string[]">
  URL or relative path to your OpenAPI spec file(s).
  <CodeGroup>
    ```json Single file
    "openapi": "/api-reference/openapi.json"
    ```
    ```json Multiple files
    "openapi": ["/openapi-v1.json", "/openapi-v2.json"]
    ```
  </CodeGroup>
</ResponseField>

## SEO

<ResponseField name="seo" type="object">
  Search engine optimization settings.
  <Expandable title="SEO">
    <ResponseField name="metatags" type="object">
      Additional `<meta>` tags as key-value pairs.
    </ResponseField>
    <ResponseField name="indexing" type="string">
      `"navigable"` to only index pages listed in navigation.
    </ResponseField>
  </Expandable>
</ResponseField>

## Search and contextual menus

<ResponseField name="search" type="object">
  <Expandable title="Search">
    <ResponseField name="prompt" type="string">Placeholder text in the search bar.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="contextual" type="object">
  Controls what appears in the right-click context menu on code blocks.
  <Expandable title="Contextual">
    <ResponseField name="options" type="string[]">
      Array of enabled options. Values: `"copy"`, `"view"`, `"chatgpt"`, `"claude"`, `"cursor"`, `"mcp"`.
    </ResponseField>
  </Expandable>
</ResponseField>

## Other settings

<ResponseField name="styling" type="object">
  <Expandable title="Styling">
    <ResponseField name="eyebrows" type="string">`"section"` to show section labels above page titles.</ResponseField>
    <ResponseField name="latex" type="boolean">Enable LaTeX math rendering.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="icons" type="object">
  <Expandable title="Icons">
    <ResponseField name="library" type="string">Icon library to use. Default: `"fontawesome"`. Options: `"fontawesome"`, `"lucide"`.</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  URL or path to a background image displayed behind every page.
</ResponseField>
